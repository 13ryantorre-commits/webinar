    <?php
    
    $FullName = "Baba Cops"; 

    // 2. Gagamit tayo ng explode() para hatiin ang pangalan base sa "space"
    // Gagawin nitong array ang pangalan: ["Temple", "Joy"]
    $name_parts = explode(" ", $FullName);

    // 3. Gagamit tayo ng end() para makuha ang pinakahuling item sa array (ang Last Name)
    $LastName = end($name_parts);

    echo "<h1>Welcome to My Inventory System</h1>";
    
    // I-display natin ang Last Name sa baba
    echo "<p>User: <b>" . $LastName . "</b></p>";

    ?>
